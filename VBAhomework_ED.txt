Sub StockVolume2014()' Set an initial variable for holding the ticker name  Dim ws As Worksheet  Dim tickername As String  For Each ws In Worksheets          ws.Cells(1, 11).Value = "Ticker"        ws.Cells(1, 12).Value = "Total Stock Volume"          ws.Activate  ' Set an initial variable for holding the total per ticker symbol  Dim TotalStockVolume As Variant  TotalStockVolume = 0  ' Keep track of the location for each ticker symbol in the summary table  Dim currentSummaryRow As Integer  currentSummaryRow = 2  Dim lastRow As Variant  lastRow = Cells(Rows.Count, 1).End(xlUp).Row  ' Loop through all ticker symbols  For i = 2 To lastRow    ' Add to the ticker Total    TotalStockVolume = TotalStockVolume + Cells(i, 7).Value    ' Check if if the next row has a different name as the current ticker    ' if so, we're on the last row for this ticker    If Cells(i + 1, 1).Value <> Cells(i, 1).Value Then      ' Set the ticker name      tickername = Cells(i, 1).Value      ' Print this ticker's info in the summary table      Cells(currentSummaryRow, 11).Value = tickername      Cells(currentSummaryRow, 12).Value = TotalStockVolume            ' Add one to the summary table row      currentSummaryRow = currentSummaryRow + 1      ' Reset the Ticker Total      TotalStockVolume = 0       End If    Next i    Next wsEnd Sub
